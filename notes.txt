
SELECT `student mail`, results.`question number`, question, `answer submitted 1`, `answer submitted 2`, `answer submitted 3`, `answer submitted 4`, `question result`
FROM results, quizz
WHERE results.`quizz title` = quizz.title
AND results.`question number` = quizz.`question number`
AND results.`quizz title` = "QCM DE DEMO"
AND results.teacher = 1
AND results.`student group` = "BTS SN1"
AND results.date = "2023-03-20 10:46:45"
ORDER BY results.`question number`

SELECT MAX(`quizz number`) FROM quizz;
